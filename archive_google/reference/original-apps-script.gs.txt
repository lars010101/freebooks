/**
 * Complete Financial Reporting System
 *
 * ONE FILE — paste this into Apps Script (replacing everything).
 *
 * Setup: Run createFullSystem first, then createStagingSystem, then setupControls.
 * Or just run: initializeEverything()
 *
 * Tabs created:
 *   CONFIG, COA, Profit & Loss, Balance Sheet, Cash Flow,
 *   Changes in Equity, General Ledger, Integrity Checks,
 *   MAPPING, STAGING
 *
 * Menu: ⚖️ Accounting (appears on sheet open)
 */

function initializeEverything() {
  SpreadsheetApp.getActiveSpreadsheet().toast("Initializing system… please wait.", "⏳ Loading", -1);
  // 1. Foundation tabs (no formula dependencies)
  createConfigTab();
  createCOATab();
  createLocksTab();
  createStagingSystem();
  setupControls();
  
  // 2. Build SUMMARY from JOURNAL (must exist before report formulas)
  buildSummary_();
  
  // 3. Now create report tabs (their SUMIFS reference SUMMARY which exists)
  createReportTabs_();
  SpreadsheetApp.getUi().alert(
    "✅ System initialized.\n\n" +
    "Tabs: CONFIG, COA, LOCKS, P&L, Balance Sheet, Cash Flow, Changes in Equity, GL, Integrity Checks, MAPPING, STAGING, SUMMARY, CF_CACHE\n\n" +
    "Menu: ⚖️ Accounting (reload sheet to see it)"
  );
}

  createChangesInEquityV3();
  createGeneralLedgerV3();
  createIntegrityChecksV5();

  var ss = SpreadsheetApp.getActiveSpreadsheet();
  for (var i = 0; i < REPORT_SHEETS_.length; i++) {

    if (sheet) {

function deleteAndCreate_(ss, name) {
  var existing = ss.getSheetByName(name);
  if (existing) ss.deleteSheet(existing);
  return ss.insertSheet(name);
}

function fmt_(sheet, range) {
  sheet.getRange(range).setNumberFormat('#,##0.00;(#,##0.00)');
}

function bold_(sheet, cell, value) {
  sheet.getRange(cell).setValue(value).setFontWeight("bold");
}

function separator_(sheet, row) {
  sheet.getRange(row + ":" + row).setBackground("#eeeeee");
}

function sectionHeader_(sheet, cell, value) {
  sheet.getRange(cell).setValue(value).setFontWeight("bold").setFontSize(11);
}

function totalRow_(sheet, row, lastCol, style) {
  var bs = (style === "heavy") ? SpreadsheetApp.BorderStyle.SOLID_MEDIUM : SpreadsheetApp.BorderStyle.SOLID;
  sheet.getRange("A" + row + ":" + lastCol + row).setBorder(true, null, true, null, null, null, "#000000", bs);
}

// Debit/Credit balance expressions (coerce text to number)
var DB = '(IF(JOURNAL!$E$2:$E$9999="",0,JOURNAL!$E$2:$E$9999*1))-(IF(JOURNAL!$F$2:$F$9999="",0,JOURNAL!$F$2:$F$9999*1))';
var CR = '(IF(JOURNAL!$F$2:$F$9999="",0,JOURNAL!$F$2:$F$9999*1))-(IF(JOURNAL!$E$2:$E$9999="",0,JOURNAL!$E$2:$E$9999*1))';

/**
 * Generates a SUMPRODUCT formula for a P&L line item.
 * References the account name in column A of the current row.
 * cfgRow: CONFIG row for the FY (11=current, 12=FY-1, etc.)
 * valExpr: DB or CR
 * acctRef: cell reference for account name, e.g. "$A7"
 */
function plFormula_(cfgRow, acctRef, valExpr) {
  var sumCol = (valExpr === CR) ? 'F' : 'E';
  return '=IF(' + acctRef + '="","",IFERROR(SUMIFS(SUMMARY!$' + sumCol + ':$' + sumCol + ',SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$C:$C,' + acctRef + '),0))';
}

function bsFormula_(cfgRow, acctRef, valExpr) {
  var sumCol = (valExpr === DB) ? 'G' : 'H';
  return '=IF(' + acctRef + '="","",IFERROR(SUMIFS(SUMMARY!$' + sumCol + ':$' + sumCol + ',SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$C:$C,' + acctRef + '),0))';
}

/**
 * Fill per-cell formulas for a section.
 * sheet: the sheet
 * startRow: first data row
 * numRows: how many rows to fill
 * formulaFn: function(cfgRow, acctRef) -> formula string
 */
function fillYearFormulas_(sheet, startRow, numRows, formulaFn) {
  for (var r = 0; r < numRows; r++) {
    var row = startRow + r;
    var acctRef = '$A$' + row;
    for (var i = 0; i < 5; i++) {
      var col = String.fromCharCode(66 + i); // B, C, D, E, F
      var cfgRow = 11 + i;
      sheet.getRange(col + row).setFormula(formulaFn(cfgRow, acctRef));
    }
  }
}

function createCOATab() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = deleteAndCreate_(ss, "COA");
  
  sheet.getRange("A1").setValue("Account").setFontWeight("bold");
  sheet.getRange("B1").setValue("Company").setFontWeight("bold");
  sheet.getRange("C1").setValue("Type").setFontWeight("bold");
  sheet.getRange("D1").setValue("Code").setFontWeight("bold");
  sheet.getRange("E1").setValue("Name").setFontWeight("bold");
  sheet.getRange("F1").setValue("CF Category").setFontWeight("bold");
  
  var journal = ss.getSheetByName("JOURNAL");
  if (!journal) { sheet.getRange("A2").setValue("ERROR: JOURNAL not found"); return; }
  
  var data = journal.getDataRange().getValues();
  var accounts = {};
  for (var i = 1; i < data.length; i++) {
    var company = String(data[i][0]).trim();
    var account = String(data[i][3]).trim();
    if (!company || !account) continue;
    var key = company + "|" + account;
    if (!accounts[key]) accounts[key] = { company: company, account: account };
  }
  
  var sorted = Object.values(accounts).sort(function(a, b) {
    if (a.company < b.company) return -1;
    if (a.company > b.company) return 1;
    return a.account < b.account ? -1 : 1;
  });
  
  var output = [];
  for (var i = 0; i < sorted.length; i++) {
    var co = sorted[i].company;
    var acct = sorted[i].account;
    var code = acct.split(" ")[0];
    var name = acct.substring(code.length).trim();
    var num = parseInt(code) || 0;
    var first = code.charAt(0);
    
    var type = classifyAccount_(co, code);
    var cfCat = classifyCfCategory_(co, code, type);
    output.push([acct, co, type, code, name, cfCat]);
  }
  
  if (output.length > 0) sheet.getRange(2, 1, output.length, 6).setValues(output);
  
  var typeRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(["Asset", "Liability", "Equity", "Revenue", "Expense", "Closing"], true)
    .setAllowInvalid(false).build();
  sheet.getRange("C2:C" + (output.length + 1)).setDataValidation(typeRule);
  
  var cfRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(["Cash", "Op-WC", "Investing", "Financing", ""], true)
    .setAllowInvalid(true).build();
  sheet.getRange("F2:F" + (output.length + 1)).setDataValidation(cfRule);
  
  sheet.getRange("A1:F1").setBackground("#4472C4").setFontColor("#FFFFFF");
  sheet.setColumnWidth(1, 400); sheet.setColumnWidth(2, 280);
  sheet.setColumnWidth(3, 100); sheet.setColumnWidth(4, 80); sheet.setColumnWidth(5, 300);
  sheet.setColumnWidth(6, 110);
  
  SpreadsheetApp.flush();
  Logger.log("✅ COA: " + output.length + " accounts.");
}

/**
 * Auto-classify CF Category based on account type and code.
 * Revenue/Expense/Closing get blank (flow through Net Income).
 * BS accounts get classified into Cash/Op-WC/Investing/Financing.
 * P&L accounts (Revenue/Expense/Closing) get blank — they flow through Net Income.
 */
function classifyCfCategory_(company, code, type) {
  // P&L and Closing accounts flow through Net Income — no CF category needed
  if (type === "Revenue" || type === "Expense" || type === "Closing") return "";
  
  var num = parseInt(code) || 0;
  
  if (company === "Inteligo") {
    // Cash accounts
    if (/^101414|^191/.test(code)) return "Cash";
    // Investment accounts → Investing
    if (/^101220|^101229|^1350|^194[0-9]/.test(code)) return "Investing";
    // Trade receivables → Op-WC
    if (/^100010|^1510/.test(code)) return "Op-WC";
    // Other receivables (dividends, interest, etc.) → Op-WC
    if (/^10121[0-9]|^10122[1-8]|^165/.test(code)) return "Op-WC";
    // Tax accounts → Op-WC
    if (/^201020|^2510|^164|^163|^265|^264/.test(code)) return "Op-WC";
    // Trade payables & accruals → Op-WC
    if (/^200|^2990|^244|^249/.test(code)) return "Op-WC";
    // Borrowings → Financing
    if (/^201130|^239300|^2490|^2893/.test(code)) return "Financing";
    // Share capital → Financing
    if (/^203080/.test(code)) return "Financing";
    // Dividends → Financing
    if (/^203040|^201110|^2898/.test(code)) return "Financing";
    // Retained earnings → blank (closing entry transfer, already captured via Net Income)
    if (/^203070/.test(code)) return "";
  } else {
    // Owner (Swedish BAS)
    if (/^1930$|^1940$|^198/.test(code)) return "Cash";
    if (/^193[^0]/.test(code)) return "Cash";
    if (/^1350/.test(code)) return "Investing";
    if (/^15/.test(code)) return "Op-WC";
    if (/^16/.test(code)) return "Op-WC";
    if (/^24|^25|^26/.test(code)) return "Op-WC";
    if (/^2081/.test(code)) return "Financing";
    // Retained earnings → blank (closing entry transfer, already captured via Net Income)
    if (/^2091|^2098|^2099/.test(code)) return "";
    if (/^2893|^2898/.test(code)) return "Financing";
  }
  
  // Fallback: unclassified BS account — leave blank (will be caught by coverage check)
  return "";
}


// ═══════════════════════════════════════════════════
// PROFIT & LOSS — 5-year, per-cell SUMPRODUCT
// ═══════════════════════════════════════════════════

function createProfitAndLossV3() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = deleteAndCreate_(ss, "Profit & Loss");
  
  bold_(sheet, "A1", "Company");
  sheet.getRange("B1").setFormula("=CONFIG!B1").setFontWeight("bold");
  bold_(sheet, "A2", "Currency");
  sheet.getRange("B2").setFormula("=CONFIG!B5");
  
  // Year headers
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + "3").setFormula('="FY" & CONFIG!A' + cfgRow).setFontWeight("bold").setHorizontalAlignment("right");
    sheet.getRange(col + "4").setFormula('=TEXT(CONFIG!B' + cfgRow + ',"mmm yy") & "–" & TEXT(CONFIG!C' + cfgRow + ',"mmm yy")').setFontColor("#666666").setHorizontalAlignment("right");
  }
  separator_(sheet, 5);
  
  // ── REVENUE ──
  var REV_ROWS = 12; // max revenue accounts
  var revStart = 7;
  var revTotalRow = revStart + REV_ROWS;  // row 19
  var expHeaderRow = revTotalRow + 2;      // row 21
  
  sectionHeader_(sheet, "A6", "REVENUE");
  
  // Account list from COA (filtered by company + type=Revenue)
  var revQuery = '=IFERROR(QUERY({COA!A$2:A, COA!B$2:B, COA!C$2:C}, '
    + '"SELECT Col1 WHERE Col2=\'"&CONFIG!B1&"\' AND Col3=\'Revenue\' ORDER BY Col1 LABEL Col1 \'\'"), '
    + '{"(No revenue accounts)"})';
  sheet.getRange("A" + revStart).setFormula(revQuery);
  
  // Per-cell SUMPRODUCT for each revenue account × each year
  // Revenue = Credit - Debit (shown as positive when credit > debit)
  fillYearFormulas_(sheet, revStart, REV_ROWS, function(cfgRow, acctRef) {
    return plFormula_(cfgRow, acctRef, CR);
  });
  
  // Total Revenue
  bold_(sheet, "A" + revTotalRow, "TOTAL REVENUE");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + revTotalRow).setFormula(
      '=IFERROR(SUMIFS(SUMMARY!$F:$F,SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$D:$D,"Revenue"),0)');
    sheet.getRange(col + revTotalRow).setFontWeight("bold");
  }
  totalRow_(sheet, revTotalRow, "F");
  separator_(sheet, revTotalRow + 1);
  
  // ── EXPENSES ──
  var EXP_ROWS = 20; // max expense accounts
  var expStart = expHeaderRow + 1;         // row 22
  var expTotalRow = expStart + EXP_ROWS;   // row 42
  
  sectionHeader_(sheet, "A" + expHeaderRow, "EXPENSES");
  
  var expQuery = '=IFERROR(QUERY({COA!A$2:A, COA!B$2:B, COA!C$2:C}, '
    + '"SELECT Col1 WHERE Col2=\'"&CONFIG!B1&"\' AND Col3=\'Expense\' ORDER BY Col1 LABEL Col1 \'\'"), '
    + '{"(No expense accounts)"})';
  sheet.getRange("A" + expStart).setFormula(expQuery);
  
  // Expenses = Debit - Credit (shown as positive when debit > credit)
  fillYearFormulas_(sheet, expStart, EXP_ROWS, function(cfgRow, acctRef) {
    return plFormula_(cfgRow, acctRef, DB);
  });
  
  // Total Expenses
  bold_(sheet, "A" + expTotalRow, "TOTAL EXPENSES");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + expTotalRow).setFormula(
      '=IFERROR(SUMIFS(SUMMARY!$E:$E,SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$D:$D,"Expense"),0)');
    sheet.getRange(col + expTotalRow).setFontWeight("bold");
  }
  totalRow_(sheet, expTotalRow, "F");
  separator_(sheet, expTotalRow + 1);
  
  // ── NET PROFIT / (LOSS) ──
  var netRow = expTotalRow + 2;
  sectionHeader_(sheet, "A" + netRow, "NET PROFIT / (LOSS)");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    sheet.getRange(col + netRow).setFormula('=' + col + revTotalRow + '-' + col + expTotalRow);
    sheet.getRange(col + netRow).setFontWeight("bold").setFontSize(11);
  }
  totalRow_(sheet, netRow, "F", "heavy");
  
  // Named ranges
  ss.setNamedRange("PL_NetIncome", sheet.getRange("B" + netRow));
  
  // Formatting
  sheet.getRange("B" + revStart + ":F" + expTotalRow).setNumberFormat('#,##0.00;(#,##0.00)');
  sheet.getRange("B" + netRow + ":F" + netRow).setNumberFormat('#,##0.00;(#,##0.00)');
  sheet.setColumnWidth(1, 400);
  for (var i = 0; i < 5; i++) sheet.setColumnWidth(2 + i, 130);
  
  SpreadsheetApp.flush();
}

// BALANCE SHEET — 5-year, per-cell SUMPRODUCT
// ═══════════════════════════════════════════════════

function createBalanceSheetV3() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = deleteAndCreate_(ss, "Balance Sheet");
  
  bold_(sheet, "A1", "Company");
  sheet.getRange("B1").setFormula("=CONFIG!B1").setFontWeight("bold");
  bold_(sheet, "A2", "Currency");
  sheet.getRange("B2").setFormula("=CONFIG!B5");
  
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + "3").setFormula('="As at " & TEXT(CONFIG!C' + cfgRow + ',"d mmm yyyy")').setFontWeight("bold").setHorizontalAlignment("right").setFontSize(9);
  }
  separator_(sheet, 4);
  
  // ── ASSETS ──
  var ASSET_ROWS = 14;
  var assetStart = 6;
  var assetTotalRow = assetStart + ASSET_ROWS; // 20
  
  sectionHeader_(sheet, "A5", "ASSETS");
  
  sheet.getRange("A" + assetStart).setFormula(
    '=IFERROR(QUERY({COA!A$2:A, COA!B$2:B, COA!C$2:C}, "SELECT Col1 WHERE Col2=\'"&CONFIG!B1&"\' AND Col3=\'Asset\' ORDER BY Col1 LABEL Col1 \'\'"), {"(No assets)"})');
  
  // Assets: debit balance (cumulative)
  fillYearFormulas_(sheet, assetStart, ASSET_ROWS, function(cfgRow, acctRef) {
    return bsFormula_(cfgRow, acctRef, DB);
  });
  
  bold_(sheet, "A" + assetTotalRow, "TOTAL ASSETS");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + assetTotalRow).setFormula(
      '=IFERROR(SUMIFS(SUMMARY!$G:$G,SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$D:$D,"Asset"),0)');
    sheet.getRange(col + assetTotalRow).setFontWeight("bold");
  }
  totalRow_(sheet, assetTotalRow, "F");
  separator_(sheet, assetTotalRow + 1);
  
  // ── LIABILITIES ──
  var LIAB_ROWS = 14;
  var liabHeaderRow = assetTotalRow + 2; // 22
  var liabStart = liabHeaderRow + 1;     // 23
  var liabTotalRow = liabStart + LIAB_ROWS; // 37
  
  sectionHeader_(sheet, "A" + liabHeaderRow, "LIABILITIES");
  
  sheet.getRange("A" + liabStart).setFormula(
    '=IFERROR(QUERY({COA!A$2:A, COA!B$2:B, COA!C$2:C}, "SELECT Col1 WHERE Col2=\'"&CONFIG!B1&"\' AND Col3=\'Liability\' ORDER BY Col1 LABEL Col1 \'\'"), {"(No liabilities)"})');
  
  // Liabilities: credit balance (cumulative)
  fillYearFormulas_(sheet, liabStart, LIAB_ROWS, function(cfgRow, acctRef) {
    return bsFormula_(cfgRow, acctRef, CR);
  });
  
  bold_(sheet, "A" + liabTotalRow, "TOTAL LIABILITIES");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + liabTotalRow).setFormula(
      '=IFERROR(SUMIFS(SUMMARY!$H:$H,SUMMARY!$A:$A,CONFIG!$B$1,SUMMARY!$B:$B,CONFIG!$A$' + cfgRow + ',SUMMARY!$D:$D,"Liability"),0)');
    sheet.getRange(col + liabTotalRow).setFontWeight("bold");
  }
  totalRow_(sheet, liabTotalRow, "F");
  separator_(sheet, liabTotalRow + 1);
  
  // ── EQUITY ──
  var EQ_ROWS = 6;
  var eqHeaderRow = liabTotalRow + 2;  // 39
  var eqStart = eqHeaderRow + 1;       // 40
  var netIncomeRow = eqStart + EQ_ROWS; // 46
  var eqTotalRow = netIncomeRow + 2;    // 48
  
  sectionHeader_(sheet, "A" + eqHeaderRow, "EQUITY");
  
  sheet.getRange("A" + eqStart).setFormula(
    '=IFERROR(QUERY({COA!A$2:A, COA!B$2:B, COA!C$2:C}, "SELECT Col1 WHERE Col2=\'"&CONFIG!B1&"\' AND Col3=\'Equity\' ORDER BY Col1 LABEL Col1 \'\'"), {"(No equity)"})');
  
  fillYearFormulas_(sheet, eqStart, EQ_ROWS, function(cfgRow, acctRef) {
    return bsFormula_(cfgRow, acctRef, CR);
  });
  
  // Cumulative unclosed P&L: Revenue(all time) - Expenses(all time) - Closing entries(all time)
  // For closed years this = 0 (no double-counting). For open years = current year P&L.
  bold_(sheet, "A" + netIncomeRow, "Unclosed P&L");
  for (var i = 0; i < 5; i++) {
    var col = String.fromCharCode(66 + i);
    var cfgRow = 11 + i;
    sheet.getRange(col + netIncomeRow).setFormula(

/**
 * Scans all BS accounts with period movements and checks which ones have
 * no CF Category assigned in COA. Outputs to a dialog.
 * Run from menu: ⚖️ Accounting → 🔍 CF Coverage Check
 */
function cfCoverageCheck() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var config = ss.getSheetByName("CONFIG");
  var coa = ss.getSheetByName("COA");
  var summary = ss.getSheetByName("SUMMARY");
  
  var company = config.getRange("B1").getValue();
  var fy = config.getRange("A11").getValue();
  
  // Build COA maps
  var coaData = coa.getDataRange().getValues();
  var typeMap = {};
  var cfCatMap = {};
  for (var i = 1; i < coaData.length; i++) {
    var acct = String(coaData[i][0]).trim();
    var co = String(coaData[i][1]).trim();
    var type = String(coaData[i][2] || "").trim();
    var cfCat = String(coaData[i][5] || "").trim();
    var key = co + "|" + acct;
    typeMap[key] = type;
    cfCatMap[key] = cfCat;
  }
  
  // Read SUMMARY for current company+FY
  var sumData = summary.getDataRange().getValues();
  var plTypes = {"Revenue": true, "Expense": true, "Closing": true};
  
  var orphans = [];
  var total = 0;
  for (var i = 1; i < sumData.length; i++) {
    var co = String(sumData[i][0]).trim();
    var sumFy = sumData[i][1];
    var acct = String(sumData[i][2]).trim();
    var periodDebitNet = parseFloat(sumData[i][4]) || 0;
    
    if (co !== company || sumFy !== fy) continue;
    if (Math.abs(periodDebitNet) < 0.005) continue;
    
    var key = co + "|" + acct;
    var type = typeMap[key] || "";
    var cfCat = cfCatMap[key] || "";
    
    // Skip P&L accounts (flow through Net Income)
    if (plTypes[type]) continue;
    
    // Skip accounts that have a CF Category
    if (cfCat) continue;
    
    // Skip Equity accounts (RE movements are closing entry transfers already in NI;
    // Share Capital/Dividends should already have Financing category)
    if (type === "Equity") continue;
    
    orphans.push({ account: acct, type: type, movement: periodDebitNet });
    total += periodDebitNet;
  }
  
  // Build output
  var msg = "CF COVERAGE CHECK — " + company + " FY" + fy + "\n\n";
  
  if (orphans.length === 0) {
    msg += "✅ All BS accounts with period movements have a CF Category in COA.\n";
  } else {
    msg += "❌ " + orphans.length + " account(s) with NO CF Category:\n\n";
    orphans.sort(function(a,b) { return Math.abs(b.movement) - Math.abs(a.movement); });
    for (var i = 0; i < orphans.length; i++) {
      var o = orphans[i];
      msg += "  " + o.account + " [" + o.type + "]  →  " + Utilities.formatString("%.2f", o.movement) + "\n";
    }
    msg += "\nTotal uncaptured: " + Utilities.formatString("%.2f", total) + "\n";
    msg += "\nFix: set CF Category in the COA sheet (column F) for these accounts,";
    msg += "\nthen run: Rebuild Summary → view Cash Flow.";
    msg += "\nValid categories: Cash, Op-Adjust, Op-WC, Investing, Financing";
  }
  
  SpreadsheetApp.getUi().alert(msg);
}

// ═══════════════════════════════════════════════════
// PROCESS STAGING — column-based auto-map
// ═══════════════════════════════════════════════════

function processStaging() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var staging = ss.getSheetByName("STAGING");
  var mapping = ss.getSheetByName("MAPPING");
  var config = ss.getSheetByName("CONFIG");
  
  if (!staging || !mapping) {
    SpreadsheetApp.getUi().alert("STAGING or MAPPING sheet not found.");
    return;
  }
  
  SpreadsheetApp.getActiveSpreadsheet().toast("Processing bank statement…", "⏳ Loading", -1);
  
  // Read input data from columns A–C (row 3 downward)
  var lastRow = staging.getLastRow();
  if (lastRow < 3) {
    SpreadsheetApp.getUi().alert("No data. Paste bank transactions in columns A–C starting at row 3.");
    return;
  }
  
  var inputData = staging.getRange(3, 1, lastRow - 2, 3).getValues(); // A:Date, B:Description, C:Amount
  
  var transactions = [];
  for (var i = 0; i < inputData.length; i++) {
    var dateVal = inputData[i][0];
    var desc = String(inputData[i][1]).trim();
    var rawAmount = inputData[i][2];
    
    // Parse amount: handle strings like "12,000.00SGD", "-2,398.00SGD", plain numbers, etc.
    var amount;
    if (typeof rawAmount === "number") {
      amount = rawAmount;
    } else {
      var amtStr = String(rawAmount).trim();
      // Remove currency codes (SGD, USD, EUR, SEK, etc.) and whitespace
      amtStr = amtStr.replace(/[A-Za-z\s]/g, "");
      // Remove thousands separators (commas) but keep decimal point
      amtStr = amtStr.replace(/,/g, "");
      amount = parseFloat(amtStr) || 0;
    }
    
    // Skip empty rows
    if (!dateVal && !desc && amount === 0) continue;
    
    // Coerce date
    if (dateVal instanceof Date) { /* ok */ }
    else if (typeof dateVal === "number") {
      dateVal = new Date(Math.round((dateVal - 25569) * 86400 * 1000));
    } else if (typeof dateVal === "string" && dateVal.trim()) {
      dateVal = new Date(dateVal.trim());
    } else {
      continue; // skip rows without a valid date
    }
    if (isNaN(dateVal.getTime())) continue;
    
    if (amount === 0) continue; // skip zero-amount rows (e.g. headers, summaries)
    
    transactions.push({ date: dateVal, description: desc, amount: amount, origRow: i });
  }
  
  // Sort by date ascending so Record IDs are assigned chronologically
  transactions.sort(function(a, b) {
    return a.date.getTime() - b.date.getTime() || a.origRow - b.origRow;
  });
  
  if (transactions.length === 0) {
    SpreadsheetApp.getUi().alert("No valid transactions found.\n\nExpected columns A–C:\n  A: Date\n  B: Description\n  C: Amount (positive = inflow, negative = outflow)");
    return;
  }
  
  // Read mapping rules
  var mapData = mapping.getDataRange().getValues();
  var rules = [];
  for (var i = 1; i < mapData.length; i++) {
    var keyword = String(mapData[i][0]).trim();
    var account = String(mapData[i][1]).trim();
    var label = String(mapData[i][2]).trim();
    if (keyword && account) {
      rules.push({ keyword: keyword.toUpperCase(), account: account, label: label });
    }
  }
  
  // Build existing journal entries for duplicate detection
  var company = config ? config.getRange("B1").getValue() : "Inteligo";
  var bankAccount = config ? config.getRange("B5").getValue() : "SGD";
  // Determine bank account name from COA (first Cash-category asset)
  var coa = ss.getSheetByName("COA");
  bankAccount = "101414 Bank"; // default
  if (coa) {
    var coaData = coa.getDataRange().getValues();
    for (var c = 1; c < coaData.length; c++) {
      if (String(coaData[c][1]).trim() === company && String(coaData[c][5]).trim() === "Cash") {
        bankAccount = String(coaData[c][0]).trim();
        break;
      }
    }
  }
  
  var journal = ss.getSheetByName("JOURNAL");
  var existingTxns = {};
  if (journal) {
    var jData = journal.getDataRange().getValues();
    for (var j = 1; j < jData.length; j++) {
      var jCo = String(jData[j][0]).trim();
      var jDate = jData[j][2];
      var jAcct = String(jData[j][3]).trim();
      var jDebit = parseFloat(jData[j][4]) || 0;
      var jCredit = parseFloat(jData[j][5]) || 0;
      if (jCo === company && jAcct === bankAccount && jDate instanceof Date) {
        var jAmt = jDebit - jCredit;
        var dupKey = jDate.toISOString().slice(0,10) + "|" + jAmt.toFixed(2);
        if (!existingTxns[dupKey]) existingTxns[dupKey] = 0;
        existingTxns[dupKey]++;
      }
    }
  }
  
  // Generate Record IDs: BNK/YYYY/NNNNN
  var maxId = 0;
  if (journal) {
    var ids = journal.getRange("B2:B" + journal.getLastRow()).getValues();
    for (var j = 0; j < ids.length; j++) {
      var idStr = String(ids[j][0]);
      var idMatch = /BNK\d?\/\d{4}\/(\d+)/.exec(idStr);
      if (idMatch) {
        var num = parseInt(idMatch[1]);
        if (num > maxId) maxId = num;
      }
    }
  }
  
  var dupCount = 0;
  var mappedCount = 0;
  var unmappedCount = 0;
  
  // Assign Record IDs in date order (transactions already sorted by date)
  for (var t = 0; t < transactions.length; t++) {
    var txn = transactions[t];
    maxId++;
    var year = txn.date.getFullYear();
    txn.recordId = "BNK/" + year + "/" + String(maxId).padStart(5, "0");
    
    // Match against mapping rules (first match wins)
    var contraAccount = "";
    var label = "";
    var status = "⚠️ UNMAPPED";
    var descUpper = txn.description.toUpperCase();
    
    for (var r = 0; r < rules.length; r++) {
      if (descUpper.indexOf(rules[r].keyword) >= 0) {
        contraAccount = rules[r].account;
        label = rules[r].label;
        status = "✅ MAPPED";
        mappedCount++;
        break;
      }
    }
    if (status === "⚠️ UNMAPPED") unmappedCount++;
    
    // Check for duplicates
    var dupKey = txn.date.toISOString().slice(0,10) + "|" + txn.amount.toFixed(2);
    if (existingTxns[dupKey]) {
      status = "⚠️ DUPLICATE?";
      dupCount++;
    }
    
    txn.output = [status, txn.date, txn.description, txn.amount, contraAccount, label, txn.recordId];
  }
  
  // Re-sort to original paste order for display
  transactions.sort(function(a, b) { return a.origRow - b.origRow; });
  
  var output = [];
  for (var t = 0; t < transactions.length; t++) {
    output.push(transactions[t].output);
  }
  
  // Clear old output (E3 downward)
  var lastOutputRow = Math.max(staging.getLastRow(), 3);
  if (lastOutputRow > 2) {
    staging.getRange("E3:K" + lastOutputRow).clearContent();
    staging.getRange("E3:K" + lastOutputRow).clearFormat();
  }
  
  // Write output
  if (output.length > 0) {
    staging.getRange(3, 5, output.length, 7).setValues(output);
    
    // Format
    staging.getRange(3, 6, output.length, 1).setNumberFormat("d mmm yyyy"); // Date
    staging.getRange(3, 8, output.length, 1).setNumberFormat("#,##0.00");   // Amount
    
    // Color code status
    for (var o = 0; o < output.length; o++) {
      var statusCell = staging.getRange(3 + o, 5);
      if (output[o][0] === "✅ MAPPED") {
        statusCell.setBackground("#E8F5E9");
      } else if (output[o][0].indexOf("DUPLICATE") >= 0) {
        statusCell.setBackground("#FFCDD2");
      } else {
        statusCell.setBackground("#FFF3E0");
      }
    }
  }
  
  var msg = "✅ Processed " + transactions.length + " transactions.\n\n"
    + "  ✅ " + mappedCount + " mapped\n"
    + "  ⚠️ " + unmappedCount + " unmapped\n"
    + "  ⚠️ " + dupCount + " potential duplicates\n\n"
    + "Review in columns E–K.\n"
    + "Fix unmapped entries (column I). Change DUPLICATE? status as needed.\n"
    + "Then: ⚖️ Accounting → 📤 Post Staged Entries";
  
  SpreadsheetApp.getActiveSpreadsheet().toast("Done.", "✅", 3);
  SpreadsheetApp.getUi().alert(msg);
}

    { name: "Example Company SE", startMonth: 1, offset: 0 }
  ];
  
  var rows = [];
  for (var c = 0; c < companies.length; c++) {
    var co = companies[c];
    for (var fy = 2016; fy <= 2027; fy++) {
      var pStart, pEnd;
      if (co.startMonth === 1) {
        pStart = new Date(fy - co.offset, 0, 1);
      } else {
        pStart = new Date(fy - co.offset - 1, co.startMonth - 1, 1);
      }
      pEnd = new Date(pStart.getFullYear(), pStart.getMonth() + 12, 0); // last day of 12th month
      rows.push([co.name, fy, pStart, pEnd, false, ""]);
    }
  }
  
  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, 6).setValues(rows);
    // Checkboxes for Locked column
    var lockRule = SpreadsheetApp.newDataValidation().requireCheckbox().build();
    sheet.getRange(2, 5, rows.length, 1).setDataValidation(lockRule);
  }
  
  sheet.getRange("C2:D" + (rows.length + 1)).setNumberFormat("d mmm yyyy");
  sheet.getRange("F2:F" + (rows.length + 1)).setNumberFormat("d mmm yyyy");
  sheet.setColumnWidth(1, 280); sheet.setColumnWidth(2, 50);
  sheet.setColumnWidth(3, 110); sheet.setColumnWidth(4, 110);
  sheet.setColumnWidth(5, 70); sheet.setColumnWidth(6, 110);
  
  SpreadsheetApp.flush();
}

function assertDateAllowed_(entryDate) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var locks = ss.getSheetByName("LOCKS");
  var config = ss.getSheetByName("CONFIG");
  if (!locks || !config) return;
  
  var company = config.getRange("B1").getValue();
  var data = locks.getDataRange().getValues();
  var entry = new Date(entryDate);
  entry.setHours(0, 0, 0, 0);
  
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim() !== company) continue;
    if (!data[i][4]) continue; // not locked
    var pStart = new Date(data[i][2]);
    var pEnd = new Date(data[i][3]);
    pStart.setHours(0, 0, 0, 0);
    pEnd.setHours(0, 0, 0, 0);
    if (entry >= pStart && entry <= pEnd) {
      throw new Error("❌ FY" + data[i][1] + " is locked (through " + pEnd.toISOString().slice(0, 10) + ").\nCannot post entries dated " + entry.toISOString().slice(0, 10) + ".");
    }
  }
}

function lockFY() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var locks = ss.getSheetByName("LOCKS");
  var config = ss.getSheetByName("CONFIG");
  if (!locks || !config) { ui.alert("LOCKS or CONFIG sheet not found."); return; }
  
  var company = config.getRange("B1").getValue();
  var response = ui.prompt("Lock FY — " + company, "Enter fiscal year to lock (e.g. 2026):", ui.ButtonSet.OK_CANCEL);
  if (response.getSelectedButton() !== ui.Button.OK) return;
  var fy = parseInt(response.getResponseText().trim());
  if (!fy) { ui.alert("Invalid FY."); return; }
  
  var data = locks.getDataRange().getValues();
  var found = false;
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim() === company && data[i][1] === fy) {
      if (data[i][4]) { ui.alert("FY" + fy + " is already locked."); return; }
      locks.getRange(i + 1, 5).setValue(true);
      locks.getRange(i + 1, 6).setValue(new Date());
      found = true;
      break;
    }
  }
  if (!found) { ui.alert("FY" + fy + " not found for " + company + "."); return; }
  
  buildSummary_();
  ui.alert("✅ FY" + fy + " locked for " + company + ".\nSummary cache rebuilt.");
}

function buildSummary_() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var journal = ss.getSheetByName("JOURNAL");
  var coa = ss.getSheetByName("COA");
  if (!journal || !coa) return;
  
  // Read COA types and CF categories
  var coaData = coa.getDataRange().getValues();
  var coaTypes = {};
  var coaCfCats = {};
  for (var i = 1; i < coaData.length; i++) {
    var acct = String(coaData[i][0]).trim();
    var co = String(coaData[i][1]).trim();
    var type = String(coaData[i][2]).trim();
    var cfCat = String(coaData[i][5] || "").trim();
    if (acct && co) {
      coaTypes[co + "|" + acct] = type;
      coaCfCats[co + "|" + acct] = cfCat;
    }
  }
  
  // Company FY settings
  var companies = {
    "Inteligo": { startMonth: 2, offset: 1 },
    "Example Company SE": { startMonth: 1, offset: 0 }
  };
  
  // Read all JOURNAL data
  var jData = journal.getDataRange().getValues();
  
  // Collect: per company per account per FY → {debit, credit}
  var buckets = {}; // key: co|fy|account → {debit, credit}
  var accountsByCompany = {}; // co → Set of accounts
  var fyByCompany = {}; // co → {min, max}
  
  for (var i = 1; i < jData.length; i++) {
    var co = String(jData[i][0]).trim();
    var dateVal = jData[i][2];
    var acct = String(jData[i][3]).trim();
    var debit = parseFloat(jData[i][4]) || 0;
    var credit = parseFloat(jData[i][5]) || 0;
    // Coerce date (handles text strings from CSV import, serial numbers, etc.)
    if (dateVal instanceof Date) {
      // already a Date — ok
    } else if (typeof dateVal === "number") {
      // Google Sheets serial date number
      dateVal = new Date(Math.round((dateVal - 25569) * 86400 * 1000));
    } else if (typeof dateVal === "string" && dateVal.trim()) {
      dateVal = new Date(dateVal.trim());
    } else {
      continue;
    }
    if (!co || !acct || isNaN(dateVal.getTime())) continue;
    
    // Determine FY
    var settings = companies[co];
    if (!settings) continue;
    var month = dateVal.getMonth() + 1; // 1-indexed
    var year = dateVal.getFullYear();
    var fy;
    if (settings.startMonth === 1) {
      fy = year + settings.offset;
    } else {
      fy = (month >= settings.startMonth) ? year + 1 + settings.offset : year + settings.offset;
    }
    
    var key = co + "|" + fy + "|" + acct;
    if (!buckets[key]) buckets[key] = { debit: 0, credit: 0 };
    buckets[key].debit += debit;
    buckets[key].credit += credit;
    
    if (!accountsByCompany[co]) accountsByCompany[co] = {};
    accountsByCompany[co][acct] = true;
    
    if (!fyByCompany[co]) fyByCompany[co] = { min: fy, max: fy };
    if (fy < fyByCompany[co].min) fyByCompany[co].min = fy;
    if (fy > fyByCompany[co].max) fyByCompany[co].max = fy;
  }
  
  // Build output rows: for each company, each account, each FY in range
  var output = [];
  for (var co in accountsByCompany) {
    var fyRange = fyByCompany[co];
    if (!fyRange) continue;
    var accounts = Object.keys(accountsByCompany[co]).sort();
    
    for (var a = 0; a < accounts.length; a++) {
      var acct = accounts[a];
      var type = coaTypes[co + "|" + acct] || "";
      var cumDebit = 0;
      var cumCredit = 0;
      
      for (var fy = fyRange.min; fy <= fyRange.max; fy++) {
        var key = co + "|" + fy + "|" + acct;
        var periodDebit = buckets[key] ? buckets[key].debit : 0;
        var periodCredit = buckets[key] ? buckets[key].credit : 0;
        var periodDebitNet = periodDebit - periodCredit;
        var periodCreditNet = periodCredit - periodDebit;
        cumDebit += periodDebit;
        cumCredit += periodCredit;
        var cumDebitNet = cumDebit - cumCredit;
        var cumCreditNet = cumCredit - cumDebit;
        
        var cfCat = coaCfCats[co + "|" + acct] || "";
        output.push([co, fy, acct, type, periodDebitNet, periodCreditNet, cumDebitNet, cumCreditNet, cfCat]);
      }
    }
  }
  
  // Write to SUMMARY sheet (clear, don't delete — preserves formula references)
  var summary = ss.getSheetByName("SUMMARY");
  if (!summary) {
    summary = ss.insertSheet("SUMMARY");
    summary.getRange("A1").setValue("Company");
    summary.getRange("B1").setValue("FY");
    summary.getRange("C1").setValue("Account");
    summary.getRange("D1").setValue("Type");
    summary.getRange("E1").setValue("PeriodDebitNet");
    summary.getRange("F1").setValue("PeriodCreditNet");
    summary.getRange("G1").setValue("CumDebitNet");
    summary.getRange("H1").setValue("CumCreditNet");
    summary.getRange("I1").setValue("CfCategory");
    summary.getRange("A1:I1").setFontWeight("bold").setBackground("#4472C4").setFontColor("#FFFFFF");
  } else {
    // Clear data rows but keep header
    var lastRow = summary.getLastRow();
    if (lastRow > 1) {
      summary.getRange(2, 1, lastRow - 1, 9).clearContent();
    }
  }
  
  if (output.length > 0) {
    summary.getRange(2, 1, output.length, 9).setValues(output);
  }
  
  summary.getRange("E2:H" + (output.length + 1)).setNumberFormat("#,##0.00");
  // Don't hide during init — hide on sheet open instead
  // summary.hideSheet();
  
  SpreadsheetApp.flush();
  Logger.log("✅ SUMMARY built: " + output.length + " rows.");
  
  // Also build CF_CACHE (cash-flow by direct/hybrid method)
  buildCfCache_(ss, jData, coaTypes, coaCfCats, companies);
}

/**
 * CF_CACHE — hybrid direct method.
 * Scans every journal entry that touches a Cash account.
 * For each such entry, finds the contra accounts in the same Record ID,
 * classifies them by CfCategory, and aggregates by Company | FY | CfCategory.
 * 
 * Columns: Company | FY | CfCategory | CashFlow (positive = inflow)
 * 
 * The CF statement reads from this cache via SUMIFS.
 */
function buildCfCache_(ss, jData, coaTypes, coaCfCats, companies) {
  // Group all journal lines by Company + Record ID
  var records = {}; // key: co|recordId → [{acct, debit, credit, date, fy}]
  
  for (var i = 1; i < jData.length; i++) {
    var co = String(jData[i][0]).trim();
    var recordId = String(jData[i][1]).trim();
    var dateVal = jData[i][2];
    var acct = String(jData[i][3]).trim();
    var debit = parseFloat(jData[i][4]) || 0;
    var credit = parseFloat(jData[i][5]) || 0;
    
    if (dateVal instanceof Date) { /* ok */ }
    else if (typeof dateVal === "number") { dateVal = new Date(Math.round((dateVal - 25569) * 86400 * 1000)); }
    else if (typeof dateVal === "string" && dateVal.trim()) { dateVal = new Date(dateVal.trim()); }
    else continue;
    if (!co || !acct || isNaN(dateVal.getTime())) continue;
    
    var settings = companies[co];
    if (!settings) continue;
    var month = dateVal.getMonth() + 1;
    var year = dateVal.getFullYear();
    var fy;
    if (settings.startMonth === 1) {
      fy = year + settings.offset;
    } else {
      fy = (month >= settings.startMonth) ? year + 1 + settings.offset : year + settings.offset;
    }
    
    var key = co + "|" + recordId;
    if (!records[key]) records[key] = [];
    records[key].push({ acct: acct, debit: debit, credit: credit, fy: fy, co: co });
  }
  
  // For each record, check if it contains a Cash account
  // If yes, the cash movement = net of Cash lines
  // Classify contra lines by their CfCategory
  var cfBuckets = {}; // key: co|fy|cfCat → cashFlow amount
  
  for (var key in records) {
    var lines = records[key];
    var co = lines[0].co;
    var fy = lines[0].fy;
    
    // Separate cash lines from non-cash lines
    var cashNet = 0; // positive = cash came in (debit to bank)
    var hasCash = false;
    var contraLines = [];
    
    for (var j = 0; j < lines.length; j++) {
      var cfCat = coaCfCats[co + "|" + lines[j].acct] || "";
      if (cfCat === "Cash") {
        hasCash = true;
        cashNet += lines[j].debit - lines[j].credit; // debit to bank = cash inflow
      } else {
        contraLines.push(lines[j]);
      }
    }
    
    if (!hasCash) continue; // No cash account in this record — skip
    
    // Classify contra lines by CfCategory
    // The total contra should equal -cashNet (balanced entry)
    // We allocate the cash flow proportionally based on contra CfCategory
    var contraByCategory = {};
    var totalContra = 0;
    
    for (var j = 0; j < contraLines.length; j++) {
      var contraCfCat = coaCfCats[co + "|" + contraLines[j].acct] || "";
      var contraType = coaTypes[co + "|" + contraLines[j].acct] || "";
      
      // Determine CF classification for contra account
      var cfClass;
      if (contraCfCat === "Op-WC") cfClass = "Operating";
      else if (contraCfCat === "Investing") cfClass = "Investing";
      else if (contraCfCat === "Financing") cfClass = "Financing";
      else if (contraType === "Revenue" || contraType === "Expense") cfClass = "Operating";
      else if (contraType === "Closing") cfClass = "Operating"; // closing entries
      else if (contraType === "Equity") cfClass = "Financing";
      else cfClass = "Unclassified";
      
      // The contra's credit-net = cash source (credit to revenue = cash from ops)
      var contraAmount = contraLines[j].credit - contraLines[j].debit;
      if (!contraByCategory[cfClass]) contraByCategory[cfClass] = 0;
      contraByCategory[cfClass] += contraAmount;
      totalContra += contraAmount;
    }
    
    // If contra doesn't balance to cashNet (shouldn't happen in balanced entries),
    // just use the cash amount directly allocated by contra proportions
    for (var cat in contraByCategory) {
      var bKey = co + "|" + fy + "|" + cat;
      if (!cfBuckets[bKey]) cfBuckets[bKey] = 0;
      // Use cash perspective: positive = cash inflow
      // contraByCategory gives us credit-net of contra which should = cashNet
      // But we anchor on actual cash movement, allocated proportionally
      if (Math.abs(totalContra) > 0.005) {
        cfBuckets[bKey] += cashNet * (contraByCategory[cat] / totalContra);
      } else {
        cfBuckets[bKey] += cashNet; // fallback: all to this category
      }
    }
  }
  
  // Also compute opening cash balances per FY
  // (cumulative Cash account debit-net up to end of prior FY)
  // We already have this in SUMMARY — no need to duplicate here
  
  // Build output
  var cfOutput = [];
  for (var bKey in cfBuckets) {
    var parts = bKey.split("|");
    var amount = cfBuckets[bKey];
    if (Math.abs(amount) < 0.005) continue;
    cfOutput.push([parts[0], parseInt(parts[1]), parts[2], Math.round(amount * 100) / 100]);
  }
  
  cfOutput.sort(function(a,b) {
    if (a[0] < b[0]) return -1; if (a[0] > b[0]) return 1;
    if (a[1] < b[1]) return -1; if (a[1] > b[1]) return 1;
    if (a[2] < b[2]) return -1; if (a[2] > b[2]) return 1;
    return 0;
  });
  
  // Write to CF_CACHE sheet
  var cfSheet = ss.getSheetByName("CF_CACHE");
  if (!cfSheet) {
    cfSheet = ss.insertSheet("CF_CACHE");
    cfSheet.getRange("A1").setValue("Company");
    cfSheet.getRange("B1").setValue("FY");
    cfSheet.getRange("C1").setValue("CfClass");
    cfSheet.getRange("D1").setValue("CashFlow");
    cfSheet.getRange("A1:D1").setFontWeight("bold").setBackground("#4472C4").setFontColor("#FFFFFF");
  } else {
    var lastRow = cfSheet.getLastRow();
    if (lastRow > 1) cfSheet.getRange(2, 1, lastRow - 1, 4).clearContent();
  }
  
  if (cfOutput.length > 0) {
    cfSheet.getRange(2, 1, cfOutput.length, 4).setValues(cfOutput);
  }
  cfSheet.getRange("D2:D" + (cfOutput.length + 1)).setNumberFormat("#,##0.00");
  
  SpreadsheetApp.flush();
  Logger.log("✅ CF_CACHE built: " + cfOutput.length + " rows.");
}

    var ss = SpreadsheetApp.getActiveSpreadsheet();

    var config = ss.getSheetByName("CONFIG");
    if (!locks || !config) return;
    var company = config.getRange("B1").getValue();

function reverseEntry() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var journal = ss.getSheetByName("JOURNAL");
  var config = ss.getSheetByName("CONFIG");
  
  if (!journal || !config) {
    ui.alert("JOURNAL or CONFIG sheet not found.");
    return;
  }
  
  // Must be on JOURNAL sheet
  if (ss.getActiveSheet().getName() !== "JOURNAL") {
    ui.alert("Please select a row in the JOURNAL sheet first, then click Reverse Entry.");
    return;
  }
  
  var company = config.getRange("B1").getValue();
  
  // Read Record ID from active row
  var activeRow = ss.getActiveCell().getRow();
  if (activeRow < 2) {
    ui.alert("Please select a data row (not the header).");
    return;
  }
  
  var rowCompany = String(journal.getRange(activeRow, 1).getValue()).trim();
  var targetId = String(journal.getRange(activeRow, 2).getValue()).trim();
  
  if (!targetId) {
    ui.alert("No Record ID found in the selected row.");
    return;
  }
  
  // Block if different company
  if (rowCompany !== company) {
    ui.alert("This entry belongs to " + rowCompany + ".\nCurrent company is " + company + ".\n\nSwitch company in CONFIG first.");
    return;
  }
  
  // Warn if already a reversal
  if (targetId.indexOf("REV/") === 0) {
    var proceed = ui.alert("Warning", "This entry (" + targetId + ") is already a reversal.\nReverse it anyway?", ui.ButtonSet.YES_NO);
    if (proceed !== ui.Button.YES) return;
  }
  
  // Find all lines with this Record ID and company
  var data = journal.getDataRange().getValues();
  var linesToReverse = [];
  
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim() === company && String(data[i][1]).trim() === targetId) {
      linesToReverse.push(data[i]);
    }
  }
  
  if (linesToReverse.length === 0) {
    ui.alert("No entries found for " + targetId + ".");
    return;
  }
  
  // Show what will be reversed + ask for date
  var summary = "Company: " + company + "\nRecord: " + targetId + "\n" + linesToReverse.length + " lines:\n\n";
  for (var i = 0; i < linesToReverse.length; i++) {
    var acct = String(linesToReverse[i][3]).trim();
    var debit = linesToReverse[i][4] || 0;
    var credit = linesToReverse[i][5] || 0;
    summary += "  " + acct + "  D:" + debit + "  C:" + credit + "\n";
  }
  
  var dateResponse = ui.prompt(
    "Reverse Entry — " + company,
    summary + "\nReversal date (YYYY-MM-DD, blank = today):",
    ui.ButtonSet.OK_CANCEL
  );
  
  if (dateResponse.getSelectedButton() !== ui.Button.OK) return;
  
  var reversalDateStr = dateResponse.getResponseText().trim();
  var reversalDate = reversalDateStr ? new Date(reversalDateStr) : new Date();
  if (isNaN(reversalDate.getTime())) {
    ui.alert("Invalid date.");
    return;
  }
  
  // Check locks
  try {
    assertDateAllowed_(reversalDate);
  } catch (e) {
    ui.alert(e.message);
    return;
  }
  
  // Generate reversal Record ID
  var year = reversalDate.getFullYear();
  var ids = journal.getRange("B2:B" + journal.getLastRow()).getValues();
  var maxRev = 0;
  for (var i = 0; i < ids.length; i++) {
    var m = /REV\/(\d{4})\/(\d+)/.exec(String(ids[i][0]));
    if (m && parseInt(m[1]) === year) {
      var num = parseInt(m[2]);
      if (num > maxRev) maxRev = num;
    }
  }
  var reversalId = "REV/" + year + "/" + String(maxRev + 1).padStart(4, "0");
  
  // Create reversed lines: swap debit and credit
  var reversalRows = [];
  for (var i = 0; i < linesToReverse.length; i++) {
    var orig = linesToReverse[i];
    var origDebit = parseFloat(orig[4]) || 0;
    var origCredit = parseFloat(orig[5]) || 0;
    var origAmount = parseFloat(orig[7]) || 0;
    
    reversalRows.push([
      company,                           // Company
      reversalId,                        // Record ID
      reversalDate,                      // Date
      String(orig[3]).trim(),            // Account
      origCredit || "",                  // Debit ← was Credit
      origDebit || "",                   // Credit ← was Debit
      String(orig[6]).trim(),            // Ccy
      -origAmount,                       // Amount (negated)
      orig[8],                           // Ccy Rate
      "REVERSAL of " + targetId,         // Label
      String(orig[10]).trim(),           // Reference
      String(orig[11]).trim(),           // Partner
      "Reversal"                         // Journal
    ]);
  }
  
  // Append to JOURNAL
  var lastRow = journal.getLastRow();
  journal.getRange(lastRow + 1, 1, reversalRows.length, 13).setValues(reversalRows);
  
  buildSummary_();
  ui.alert("✅ Reversal posted: " + reversalId + "\n\n" + reversalRows.length + " lines reversing " + targetId + "\nDated: " + reversalDate.toISOString().slice(0, 10));
}

// REFRESH COA — merge (add new, preserve existing)
// ═══════════════════════════════════════════════════

function refreshCOA() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var coa = ss.getSheetByName("COA");
  var journal = ss.getSheetByName("JOURNAL");
  
  if (!journal) {
    SpreadsheetApp.getUi().alert("JOURNAL sheet not found.");
    return;
  }
  
  // Read existing COA entries (preserve Type and CfCategory overrides)
  var existingTypes = {};
  var existingCfCats = {};
  if (coa) {
    var coaData = coa.getDataRange().getValues();
    for (var i = 1; i < coaData.length; i++) {
      var acct = String(coaData[i][0]).trim();
      var co = String(coaData[i][1]).trim();
      var type = String(coaData[i][2]).trim();
      var cfCat = String(coaData[i][5] || "").trim();
      if (acct && co) {
        if (type) existingTypes[co + "|" + acct] = type;
        existingCfCats[co + "|" + acct] = cfCat;
      }
    }
  }
  
  // Scan JOURNAL for all accounts
  var data = journal.getDataRange().getValues();
  var accounts = {};
  for (var i = 1; i < data.length; i++) {
    var company = String(data[i][0]).trim();
    var account = String(data[i][3]).trim();
    if (!company || !account) continue;
    var key = company + "|" + account;
    if (!accounts[key]) accounts[key] = { company: company, account: account };
  }
  
  // Build output, preserving existing Types
  var sorted = Object.values(accounts).sort(function(a, b) {
    if (a.company < b.company) return -1;
    if (a.company > b.company) return 1;
    return a.account < b.account ? -1 : 1;
  });
  
  var output = [];
  var newCount = 0;
  for (var i = 0; i < sorted.length; i++) {
    var co = sorted[i].company;
    var acct = sorted[i].account;
    var code = acct.split(" ")[0];
    var name = acct.substring(code.length).trim();
    var key = co + "|" + acct;
    
    var type;
    var cfCat;
    if (existingTypes[key]) {
      type = existingTypes[key]; // Preserve existing
      cfCat = existingCfCats[key] || "";
    } else {
      type = classifyAccount_(co, code); // Auto-classify new ones
      cfCat = classifyCfCategory_(co, code, type);
      newCount++;
    }
    
    output.push([acct, co, type, code, name, cfCat]);
  }
  
  // Recreate COA sheet with merged data
  if (coa) ss.deleteSheet(coa);
  coa = ss.insertSheet("COA");
  
  coa.getRange("A1").setValue("Account").setFontWeight("bold");
  coa.getRange("B1").setValue("Company").setFontWeight("bold");
  coa.getRange("C1").setValue("Type").setFontWeight("bold");
  coa.getRange("D1").setValue("Code").setFontWeight("bold");
  coa.getRange("E1").setValue("Name").setFontWeight("bold");
  coa.getRange("F1").setValue("CF Category").setFontWeight("bold");
  coa.getRange("A1:F1").setBackground("#4472C4").setFontColor("#FFFFFF");
  
  if (output.length > 0) coa.getRange(2, 1, output.length, 6).setValues(output);
  
  var typeRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(["Asset", "Liability", "Equity", "Revenue", "Expense", "Closing"], true)
    .setAllowInvalid(false).build();
  coa.getRange("C2:C" + (output.length + 1)).setDataValidation(typeRule);
  
  var cfRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(["Cash", "Op-WC", "Investing", "Financing", ""], true)
    .setAllowInvalid(true).build();
  coa.getRange("F2:F" + (output.length + 1)).setDataValidation(cfRule);
  
  coa.setColumnWidth(1, 400); coa.setColumnWidth(2, 280);
  coa.setColumnWidth(3, 100); coa.setColumnWidth(4, 80); coa.setColumnWidth(5, 300);
  coa.setColumnWidth(6, 110);
  
  SpreadsheetApp.flush();
  SpreadsheetApp.getUi().alert(
    "✅ COA refreshed.\n\n" +
    output.length + " total accounts.\n" +
    newCount + " new accounts added (auto-classified).\n" +
    (output.length - newCount) + " existing accounts preserved."
  );
}

/**
 * Auto-classify account by company and code prefix.
 */
function classifyAccount_(company, code) {
  var num = parseInt(code) || 0;
  var first = code.charAt(0);
  
  if (company === "Inteligo") {
    if (code === "999999") return "Closing";
    if (first === "1") return "Asset";
    if (code === "203040" || code === "203070" || code === "203080") return "Equity";
    if (first === "2") return "Liability";
    if (first === "3") return "Revenue";
    return "Expense";
  } else {
    // Swedish BAS chart
    if (code === "8999") return "Closing";
    if (first === "1") return "Asset";
    if (num >= 2080 && num <= 2099) return "Equity";
    if (first === "2") return "Liability";
    if (first === "3") return "Revenue";
    if (first >= "4" && first <= "7") return "Expense";
    if (num >= 8000 && num < 8400) return "Revenue";
    if (num >= 8400 && num < 8999) return "Expense";
    return "Expense";
  }
}

function postManualEntry(dateStr, label, ref, lines) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var journal = ss.getSheetByName("JOURNAL");
  var config = ss.getSheetByName("CONFIG");
  
  if (!journal) throw new Error("JOURNAL sheet not found.");
  
  // Check lock date
  assertDateAllowed_(dateStr);
  
  var company = config ? config.getRange("B1").getValue() : "Inteligo";
  var ccy = config ? config.getRange("B5").getValue() : "SGD";
  var date = new Date(dateStr);
  
  // Generate Record ID: MISC/YYYY/NNNNN
  var year = date.getFullYear();
  var ids = journal.getRange("B2:B" + journal.getLastRow()).getValues();
  var maxMisc = 0;
  for (var i = 0; i < ids.length; i++) {
    var m = /MISC\/(\d{4})\/(\d+)/.exec(String(ids[i][0]));
    if (m && parseInt(m[1]) === year) {
      var num = parseInt(m[2]);
      if (num > maxMisc) maxMisc = num;
    }
  }
  var recordId = "MISC/" + year + "/" + String(maxMisc + 1).padStart(4, "0");
  
  var rows = [];
  for (var i = 0; i < lines.length; i++) {
    var debit = lines[i].debit || "";
    var credit = lines[i].credit || "";
    var amount = (lines[i].debit || 0) - (lines[i].credit || 0);
    rows.push([
      company, recordId, date, lines[i].account,
      debit, credit, ccy, amount, 1,
      label, ref, "", "Miscellaneous Operations"
    ]);
  }
  
  var lastRow = journal.getLastRow();
  journal.getRange(lastRow + 1, 1, rows.length, 13).setValues(rows);
  
  buildSummary_();
  return "✅ Posted " + recordId + " (" + rows.length + " lines)";
}

function postToJournal() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var staging = ss.getSheetByName("STAGING");
  var journal = ss.getSheetByName("JOURNAL");
  var config = ss.getSheetByName("CONFIG");
  var ui = SpreadsheetApp.getUi();
  
  if (!staging || !journal) {
    ui.alert("STAGING or JOURNAL sheet not found.");
    return;
  }
  
  var company = config ? config.getRange("B1").getValue() : "Inteligo";
  var ccy = config ? config.getRange("B5").getValue() : "SGD";
  var journalName = "Bank Statement Import";
  
  // Determine bank account from COA (first Cash-category account for this company)
  var bankAccount = "101414 Bank";
  var coa = ss.getSheetByName("COA");
  if (coa) {
    var coaData = coa.getDataRange().getValues();
    for (var c = 1; c < coaData.length; c++) {
      if (String(coaData[c][1]).trim() === company && String(coaData[c][5]).trim() === "Cash") {
        bankAccount = String(coaData[c][0]).trim();
        break;
      }
    }
  }
  
  var lastRow = staging.getLastRow();
  if (lastRow < 3) { ui.alert("No entries to post."); return; }
  
  var data = staging.getRange(3, 5, lastRow - 2, 7).getValues(); // E:Status, F:Date, G:Desc, H:Amount, I:Account, J:Label, K:RecordID
  
  var toPost = [];
  var unmappedCount = 0;
  var skippedCount = 0;
  var postedCount = 0;
  var blockedDates = [];
  
  for (var i = 0; i < data.length; i++) {
    var status = String(data[i][0]).trim();
    var date = data[i][1];
    var desc = String(data[i][2]).trim();
    var amount = parseFloat(data[i][3]) || 0;
    var contraAccount = String(data[i][4]).trim();
    var label = String(data[i][5]).trim();
    var recordId = String(data[i][6]).trim();
    
    if (!date || !recordId || amount === 0) continue;
    
    if (status.indexOf("POSTED") >= 0 || status.indexOf("SKIP") >= 0) {
      if (status.indexOf("POSTED") >= 0) postedCount++;
      else skippedCount++;
      continue;
    }
    
    if (!contraAccount) { unmappedCount++; continue; }
    
    // Check locks
    try {
      assertDateAllowed_(date);
    } catch (e) {
      blockedDates.push(new Date(date).toISOString().slice(0, 10) + " — " + desc.substring(0, 40));
      continue;
    }
    
    var bankDebit = amount > 0 ? amount : "";
    var bankCredit = amount < 0 ? Math.abs(amount) : "";
    var contraDebit = amount < 0 ? Math.abs(amount) : "";
    var contraCredit = amount > 0 ? amount : "";
    
    toPost.push([company, recordId, date, bankAccount, bankDebit, bankCredit, ccy, amount, 1, label, "", "", journalName]);
    toPost.push([company, recordId, date, contraAccount, contraDebit, contraCredit, ccy, amount < 0 ? Math.abs(amount) : -amount, 1, label, "", "", journalName]);
  }
  
  // Duplicate detection: check existing JOURNAL for matching Company + Date + Bank Account + Amount
  var jData = journal.getDataRange().getValues();
  var existingSet = {};
  for (var j = 1; j < jData.length; j++) {
    var jCo = String(jData[j][0]).trim();
    var jDate = jData[j][2];
    var jAcct = String(jData[j][3]).trim();
    var jDebit = parseFloat(jData[j][4]) || 0;
    var jCredit = parseFloat(jData[j][5]) || 0;
    if (jCo === company && jAcct === bankAccount && jDate instanceof Date) {
      var jAmt = jDebit - jCredit;
      var key = jDate.toISOString().slice(0,10) + "|" + jAmt.toFixed(2);
      if (!existingSet[key]) existingSet[key] = 0;
      existingSet[key]++;
    }
  }
  
  var duplicates = [];
  var postingCounts = {};
  for (var i = 0; i < toPost.length; i += 2) { // step by 2 (bank line + contra line)
    var pDate = toPost[i][2];
    var pAmt = (parseFloat(toPost[i][4]) || 0) - (parseFloat(toPost[i][5]) || 0);
    if (pDate instanceof Date) {
      var key = pDate.toISOString().slice(0,10) + "|" + pAmt.toFixed(2);
      if (!postingCounts[key]) postingCounts[key] = 0;
      postingCounts[key]++;
      if (existingSet[key]) {
        duplicates.push(pDate.toISOString().slice(0,10) + " | " + pAmt.toFixed(2) + " " + ccy);
      }
    }
  }
  
  if (duplicates.length > 0) {
    var dupMsg = "⚠️ " + duplicates.length + " potential duplicate(s) found:\n\n"
      + duplicates.slice(0, 10).join("\n")
      + (duplicates.length > 10 ? "\n… and " + (duplicates.length - 10) + " more" : "")
      + "\n\nThese match existing JOURNAL entries by date + amount.\nPost anyway?";
    if (ui.alert("Duplicate Warning", dupMsg, ui.ButtonSet.YES_NO) !== ui.Button.YES) return;
  }
  
  if (blockedDates.length > 0 && toPost.length === 0) {
    ui.alert("❌ All entries fall in locked periods.\n\n" +
      "Blocked entries:\n" + blockedDates.join("\n") +
      "\n\nUnlock the FY via ⚖️ Accounting → Unlock FY.");
    return;
  }
  
  if (unmappedCount > 0 || blockedDates.length > 0) {
    var msg = "";
    if (unmappedCount > 0) msg += unmappedCount + " unmapped entries will be skipped.\n";
    if (blockedDates.length > 0) msg += blockedDates.length + " entries blocked by lock date.\n";
    msg += "\nPost the " + (toPost.length / 2) + " valid entries?";
    
    if (ui.alert("Warning", msg, ui.ButtonSet.YES_NO) !== ui.Button.YES) return;
  }
  
  if (toPost.length === 0) { ui.alert("No entries to post."); return; }
  
  var journalLastRow = journal.getLastRow();
  journal.getRange(journalLastRow + 1, 1, toPost.length, 13).setValues(toPost);
  
  // Update status
  for (var i = 0; i < data.length; i++) {
    var status = String(data[i][0]).trim();
    var contraAccount = String(data[i][4]).trim();
    if (status.indexOf("MAPPED") >= 0 && contraAccount) {
      // Check if this one was actually posted (not blocked)
      var blocked = false;
      try { assertDateAllowed_(data[i][1]); } catch (e) { blocked = true; }
      if (!blocked) {
        staging.getRange(3 + i, 5).setValue("🔒 POSTED").setBackground("#E3F2FD");
      }
    }
  }
  
  buildSummary_();
  
  var entriesPosted = toPost.length / 2;
  var msg = "✅ Posted " + entriesPosted + " transactions (" + toPost.length + " journal lines).";
  if (unmappedCount > 0) msg += "\n⚠️ " + unmappedCount + " unmapped skipped.";
  if (blockedDates.length > 0) msg += "\n🔒 " + blockedDates.length + " blocked by locked FY.";
  if (skippedCount > 0) msg += "\n❌ " + skippedCount + " marked SKIP.";
  if (postedCount > 0) msg += "\n🔒 " + postedCount + " already posted.";
  
  ui.alert(msg);
}
